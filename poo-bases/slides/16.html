$this ?

// Définition d'une classe qui représente une personne
class Personne {
    // définition des attributs / propriétés
    public $prenom;
    public $nom;
    public $age;
    public $estChanteur;

    // définition d'une méthode, 
    // une action réalisable par les objets
    public function biographie() {
        $bio = $this->prenom . ' ' . $this->nom . 
                ' est âgé de ' . $this->age . ' ans';

        if ($this->estChanteur) {
            $bio .= ' et c\'est un chanteur';
        }
        return $bio;
    }
}

$this est une variable spéciale.

$this désigne / représente l'objet en cours.

$this est utilisé uniquement dans les classes.

$this permet d'accéder aux propriétés ou aux méthodes.

Dans notre exemple, $this->prenom se lit :

"dans un objet Personne, je veux accéder à la propriété prenom".
