JS : modules

let counter = 0;
document.getElementById('button').addEventListener('click', handleCounter);
function handleCounter() {
    counter++;
    document.getElementById('counter').textContent = counter;
}
// déclaration d'un module
// un module, c'est d'abord une constante, avec un nom
const simpleCounter = {
    
    counter: 0,
    handleCounter: function() {
        // pour appeler une variable du module, on préfixe 
        // le nom de la variable par le nom du module
        simpleCounter.counter++;
    }

};
