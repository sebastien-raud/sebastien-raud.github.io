$renaud = [
    'prenom' => 'Renaud',
    'nom' => 'Séchan',
    'age' => 70,
    'estChanteur' => true
];

function biographie($personne) {
    $bio = $personne['prenom'] . ' ' . $personne['nom'] . 
           ' est âgé de ' . $personne['age'] . ' ans';

    if ($personne['estChanteur']) {
        $bio .= ' et c\'est un chanteur';
    }
    return $bio;
}

echo biographie($renaud);
// Définition d'une classe qui représente une personne
class Personne {

    // définition des attributs / propriétés
    public $prenom;
    public $nom;
    public $age;
    public $estChanteur;

}

// le mot-clé "new" permet de créer un objet
// à partir d'une classe, d'un modèle
$renaud = new Personne(); // $renaud est un objet Personne
$renaud->prenom = 'Renaud'; // accès à la propriété "prenom"
$renaud->nom = 'Séchan'; // l'opérateur -> permet d'accéder
$renaud->age = 70; // aux propriétés de l'objet
$renaud->estChanteur = true; // pour le personnaliser
