Jusque là, en programmation, nous avons utilisé :

  • // des variables de types simples
$prenom = 'Renaud'; // string
$nom = 'Séchan'; // string
$age = 70; // int, attention, c'est bientôt son anniversaire
$estChanteur = true; // boolean
  • // des variables complexes
$renaud = [
    'prenom' => 'Renaud',
    'nom' => 'Séchan',
    'age' => 70,
    'estChanteur' => true
];
  • // des fonctions
function biographie($personne) {
    $bio = $personne['prenom'] . ' ' . $personne['nom'] . ' est âgé de ' . $personne['age'] . ' ans';
    // des conditionnelles, des boucles...
    if ($personne['estChanteur']) {
        // des opérateurs (ici la concaténation)
        $bio .= ' et c\'est un chanteur';
    }
    return $bio;
}

echo biographie($renaud);
