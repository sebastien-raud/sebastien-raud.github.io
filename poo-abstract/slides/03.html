POO : abstract

Une classe abstraite peut aussi définir les méthodes abstraites.

Une méthode abstraite ne peut pas avoir de code.

Les classes filles sont obligées de définir les méthodes abstraites.

abstact class Animal {
    protected $nom;
    protected $nombrePattes;

    // méthode abstraite : elle doit être déclarée dans les classes héritées
    abstract public function communiquer();

    public function infos() {
        echo $this->nom . ' a ' . $this->nombrePattes . ' pattes.';
    }
}
class Humain extends Animal {
    public function __construct() {
        $this->nombrePattes = 2;
    }

    public function communiquer() {
        return 'parler';
    }
}
class Chien extends Animal {
    public function __construct() {
        $this->nombrePattes = 4;
    }

    public function communiquer() {
        return 'aboyer';
    }
}
