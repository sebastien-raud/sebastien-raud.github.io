$renaud = [
    'prenom' => 'Renaud',
    'nom' => 'Séchan',
    'age' => 70,
    'estChanteur' => true
];

function biographie($personne) {
    $bio = $personne['prenom'] . ' ' . $personne['nom'] . 
           ' est âgé de ' . $personne['age'] . ' ans';

    if ($personne['estChanteur']) {
        $bio .= ' et c\'est un chanteur';
    }
    return $bio;
}

echo biographie($renaud);
// Définition d'une classe qui représente une personne
class Personne {
    // définition des attributs / propriétés
    public $prenom;
    public $nom;
    public $age;
    public $estChanteur;

    // définition d'une méthode, 
    // une action réalisable par les objets
    public function biographie() {
        $bio = $this->prenom . ' ' . $this->nom . 
               ' est âgé de ' . $this->age . ' ans';

        if ($this->estChanteur) {
            $bio .= ' et c\'est un chanteur';
        }
        return $bio;
    }
}
