Classe et objets

// Définition d'une classe qui représente une personne
class Personne {

    // définition des attributs / propriétés
    public $prenom;
    public $nom;
    public $age;
    public $estChanteur;

}
// on peut créer plusieurs objets
// à partir du même modèle

// $renaud est un objet Personne
// c'est une instance de la classe Personne
$renaud = new Personne();
$renaud->prenom = 'Renaud';
$renaud->nom = 'Séchan';
$renaud->age = 70;
$renaud->estChanteur = true;

// $michael est un objet Personne
// c'est une autre instance de la classe Personne
$michael = new Personne();
$michael->prenom = 'Michael';
$michael->nom = 'Jackson';
$michael->age = 50;
$michael->estChanteur = true;
