// Définition d'une classe qui représente une personne
class Personne {
    // définition des attributs / propriétés
    public $prenom;
    public $nom;
    public $age;
    public $estChanteur;

    // définition d'une méthode, 
    // une action réalisable par les objets
    public function biographie() {
        $bio = $this->prenom . ' ' . $this->nom . 
                ' est âgé de ' . $this->age . ' ans';

        if ($this->estChanteur) {
            $bio .= ' et c\'est un chanteur';
        }
        return $bio;
    }
}
// Utilisation de la méthode

// $renaud est un objet Personne
// c'est une instance de la classe Personne
$renaud = new Personne();
$renaud->prenom = 'Renaud';
$renaud->nom = 'Séchan';
$renaud->age = 70;
$renaud->estChanteur = true;

// appel de la méthode biographie() sur
// l'objet $renaud
echo $renaud->biographie();
