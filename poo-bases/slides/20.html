Classe : méthode __construct

// Définition d'une classe qui représente une personne
class Personne {
    public $prenom;
    public $nom;
    public $age;
    public $estChanteur;

    // définition d'un constructeur avec un
    // seul paramètre obligatoire et un optionnel
    public function __construct($paramPrenom, $paramNom = null) {
        $this->prenom = $paramPrenom;
        $this->nom = $paramNom;
    }

    public function biographie() {
        $bio = $this->prenom . ' ' . $this->nom . 
                ' est âgé de ' . $this->age . ' ans';

        if ($this->estChanteur) {
            $bio .= ' et c\'est un chanteur';
        }
        return $bio;
    }
}
// Initialisation via le constructeur

// $renaud est un objet Personne.
//
// Il est initialisé avec prénom. Le nom
// a une valeur par défaut : on peut le donner
// dans le constructeur, ou non.
//
// Les paramètres sont passés au constructeur
// lors de la création de l'objet.
$renaud = new Personne('Renaud');

// les autres propriétés
$renaud->nom = 'Séchan';
$renaud->age = 70;
$renaud->estChanteur = true;

// appel de la méthode biographie() sur
// l'objet $renaud
echo $renaud->biographie();
